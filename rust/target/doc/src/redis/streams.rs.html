11/// Utility enum for passing `MAXLEN [= or ~] [COUNT]`
12/// arguments into `StreamCommands`.
13/// The enum value represents the count.
14#[derive(PartialEq, Eq, Clone, Debug, Copy)]
15pub enum StreamMaxlen {
16    /// Match an exact count
17    Equals(usize),
18    /// Match an approximate count
19    Approx(usize),
20}
21
22impl ToRedisArgs for StreamMaxlen {
23    fn write_redis_args<W>(&self, out: &mut W)
24    where
25        W: ?Sized + RedisWrite,
26    {
27        let (ch, val) = match *self {
28            StreamMaxlen::Equals(v) => ("=", v),
29            StreamMaxlen::Approx(v) => ("~", v),
30        };
31        out.write_arg(b"MAXLEN");
32        out.write_arg(ch.as_bytes());
33        val.write_redis_args(out);
34    }
35}

115/// Argument to `StreamReadOptions`
116/// Represents the Redis `GROUP <groupname> <consumername>` cmd arg.
117/// This option will toggle the cmd from `XREAD` to `XREADGROUP`
118type SRGroup = Option<(Vec<Vec<u8>>, Vec<Vec<u8>>)>;
119/// Builder options for [`xread_options`] command.
120///
121/// [`xread_options`]: ../trait.Commands.html#method.xread_options
122///
123#[derive(Default, Debug)]
124pub struct StreamReadOptions {
125    /// Set the `BLOCK <milliseconds>` cmd arg.
126    block: Option<usize>,
127    /// Set the `COUNT <count>` cmd arg.
128    count: Option<usize>,
129    /// Set the `NOACK` cmd arg.
130    noack: Option<bool>,
131    /// Set the `GROUP <groupname> <consumername>` cmd arg.
132    /// This option will toggle the cmd from XREAD to XREADGROUP.
133    group: SRGroup,
134}
135
136impl StreamReadOptions {
137    /// Indicates whether the command is participating in a group
138    /// and generating ACKs
139    pub fn read_only(&self) -> bool {
140        self.group.is_none()
141    }
142
143    /// Sets the command so that it avoids adding the message
144    /// to the PEL in cases where reliability is not a requirement
145    /// and the occasional message loss is acceptable.
146    pub fn noack(mut self) -> Self {
147        self.noack = Some(true);
148        self
149    }
150
151    /// Sets the block time in milliseconds.
152    pub fn block(mut self, ms: usize) -> Self {
153        self.block = Some(ms);
154        self
155    }
156
157    /// Sets the maximum number of elements to return per stream.
158    pub fn count(mut self, n: usize) -> Self {
159        self.count = Some(n);
160        self
161    }
162
163    /// Sets the name of a consumer group associated to the stream.
164    pub fn group<GN: ToRedisArgs, CN: ToRedisArgs>(
165        mut self,
166        group_name: GN,
167        consumer_name: CN,
168    ) -> Self {
169        self.group = Some((
170            ToRedisArgs::to_redis_args(&group_name),
171            ToRedisArgs::to_redis_args(&consumer_name),
172        ));
173        self
174    }
175}

177impl ToRedisArgs for StreamReadOptions {
178    fn write_redis_args<W>(&self, out: &mut W)
179    where
180        W: ?Sized + RedisWrite,
181    {
182        if let Some(ref group) = self.group {
183            out.write_arg(b"GROUP");
184            for i in &group.0 {
185                out.write_arg(i);
186            }
187            for i in &group.1 {
188                out.write_arg(i);
189            }
190        }
191
192        if let Some(ref ms) = self.block {
193            out.write_arg(b"BLOCK");
194            out.write_arg(format!("{ms}").as_bytes());
195        }
196
197        if let Some(ref n) = self.count {
198            out.write_arg(b"COUNT");
199            out.write_arg(format!("{n}").as_bytes());
200        }
201
202        if self.group.is_some() {
203            // noack is only available w/ xreadgroup
204            if self.noack == Some(true) {
205                out.write_arg(b"NOACK");
206            }
207        }
208    }
209}

418/// Represents a stream `id` and its field/values as a `HashMap`
419#[derive(Default, Debug, Clone)]
420pub struct StreamId {
421    /// The stream `id` (entry ID) of this particular message.
422    pub id: String,
423    /// All fields in this message, associated with their respective values.
424    pub map: HashMap<String, Value>,
425}
426
427impl StreamId {
428    /// Converts a `Value::Bulk` into a `StreamId`.
429    fn from_bulk_value(v: &Value) -> RedisResult<Self> {
430        let mut stream_id = StreamId::default();
431        if let Value::Bulk(ref values) = *v {
432            if let Some(v) = values.first() {
433                stream_id.id = from_redis_value(v)?;
434            }
435            if let Some(v) = values.get(1) {
436                stream_id.map = from_redis_value(v)?;
437            }
438        }
439
440        Ok(stream_id)
441    }
442
443    /// Fetches value of a given field and converts it to the specified
444    /// type.
445    pub fn get<T: FromRedisValue>(&self, key: &str) -> Option<T> {
446        match self.map.get(key) {
447            Some(x) => from_redis_value(x).ok(),
448            None => None,
449        }
450    }
451
452    /// Does the message contain a particular field?
453    pub fn contains_key(&self, key: &str) -> bool {
454        self.map.contains_key(key)
455    }
456
457    /// Returns how many field/value pairs exist in this message.
458    pub fn len(&self) -> usize {
459        self.map.len()
460    }
461
462    /// Returns true if there are no field/value pairs in this message.
463    pub fn is_empty(&self) -> bool {
464        self.len() == 0
465    }
466}

468type SRRows = Vec<HashMap<String, Vec<HashMap<String, HashMap<String, Value>>>>>;
469impl FromRedisValue for StreamReadReply {
470    fn from_redis_value(v: &Value) -> RedisResult<Self> {
471        let rows: SRRows = from_redis_value(v)?;
472        let keys = rows
473            .into_iter()
474            .flat_map(|row| {
475                row.into_iter().map(|(key, entry)| {
476                    let ids = entry
477                        .into_iter()
478                        .flat_map(|id_row| id_row.into_iter().map(|(id, map)| StreamId { id, map }))
479                        .collect();
480                    StreamKey { key, ids }
481                })
482            })
483            .collect();
484        Ok(StreamReadReply { keys })
485    }
486}

510type SPRInner = (
511    usize,
512    Option<String>,
513    Option<String>,
514    Vec<Option<(String, String)>>,
515);
516impl FromRedisValue for StreamPendingReply {
517    fn from_redis_value(v: &Value) -> RedisResult<Self> {
518        let (count, start, end, consumer_data): SPRInner = from_redis_value(v)?;
519
520        if count == 0 {
521            Ok(StreamPendingReply::Empty)
522        } else {
523            let mut result = StreamPendingData::default();
524
525            let start_id = start.ok_or_else(|| {
526                Error::new(
527                    ErrorKind::Other,
528                    "IllegalState: Non-zero pending expects start id",
529                )
530            })?;
531
532            let end_id = end.ok_or_else(|| {
533                Error::new(
534                    ErrorKind::Other,
535                    "IllegalState: Non-zero pending expects end id",
536                )
537            })?;
538
539            result.count = count;
540            result.start_id = start_id;
541            result.end_id = end_id;
542
543            result.consumers = consumer_data
544                .into_iter()
545                .flatten()
546                .map(|(name, pending)| StreamInfoConsumer {
547                    name,
548                    pending: pending.parse().unwrap_or_default(),
549                    ..Default::default()
550                })
551                .collect();
552
553            Ok(StreamPendingReply::Data(result))
554        }
555    }
556}

558impl FromRedisValue for StreamPendingCountReply {
559    fn from_redis_value(v: &Value) -> RedisResult<Self> {
560        let mut reply = StreamPendingCountReply::default();
561        match v {
562            Value::Bulk(outer_tuple) => {
563                for outer in outer_tuple {
564                    match outer {
565                        Value::Bulk(inner_tuple) => match &inner_tuple[..] {
566                            [Value::Data(id_bytes), Value::Data(consumer_bytes), Value::Int(last_delivered_ms_u64), Value::Int(times_delivered_u64)] =>
567                            {
568                                let id = String::from_utf8(id_bytes.to_vec())?;
569                                let consumer = String::from_utf8(consumer_bytes.to_vec())?;
570                                let last_delivered_ms = *last_delivered_ms_u64 as usize;
571                                let times_delivered = *times_delivered_u64 as usize;
572                                reply.ids.push(StreamPendingId {
573                                    id,
574                                    consumer,
575                                    last_delivered_ms,
576                                    times_delivered,
577                                });
578                            }
579                            _ => fail!((
580                                crate::types::ErrorKind::TypeError,
581                                "Cannot parse redis data (3)"
582                            )),
583                        },
584                        _ => fail!((
585                            crate::types::ErrorKind::TypeError,
586                            "Cannot parse redis data (2)"
587                        )),
588                    }
589                }
590            }
591            _ => fail!((
592                crate::types::ErrorKind::TypeError,
593                "Cannot parse redis data (1)"
594            )),
595        };
596        Ok(reply)
597    }
598}

600impl FromRedisValue for StreamInfoStreamReply {
601    fn from_redis_value(v: &Value) -> RedisResult<Self> {
602        let map: HashMap<String, Value> = from_redis_value(v)?;
603        let mut reply = StreamInfoStreamReply::default();
604        if let Some(v) = &map.get("last-generated-id") {
605            reply.last_generated_id = from_redis_value(v)?;
606        }
607        if let Some(v) = &map.get("radix-tree-nodes") {
608            reply.radix_tree_keys = from_redis_value(v)?;
609        }
610        if let Some(v) = &map.get("groups") {
611            reply.groups = from_redis_value(v)?;
612        }
613        if let Some(v) = &map.get("length") {
614            reply.length = from_redis_value(v)?;
615        }
616        if let Some(v) = &map.get("first-entry") {
617            reply.first_entry = StreamId::from_bulk_value(v)?;
618        }
619        if let Some(v) = &map.get("last-entry") {
620            reply.last_entry = StreamId::from_bulk_value(v)?;
621        }
622        Ok(reply)
623    }
624}
